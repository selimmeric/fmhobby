/*******************************************************************************
流水灯程序                                                            
 P1口连接					 
*******************************************************************************/

#include<reg52.h>
#include <intrins.h>



/*****************************************************************************
  延时函数											
******************************************************************************/

void delayms(unsigned char ms)
{
	unsigned char i;
	while(ms--)
	{
	for(i = 0; i < 120; i++);
	}
}

/*****************************************************************************
*  主程序													
******************************************************************************/
 main()
{
	unsigned char LED;
	LED = 0xfe;
	P1 = LED;
	while(1)
	{
	delayms(250);
	LED = _crol_(LED,1);		//循环右移1位，点亮下一个LED  此函数为库函数
	P1 = LED;
	}
}
